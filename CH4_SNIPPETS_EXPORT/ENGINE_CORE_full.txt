   1  /** Phone-lux calibration ("both" mode): raw sensor lux in, reference-equivalent
   2   *  lux out. Pure so the on/off paths are unit-testable; constants live in
   3   *  config.ts and are regenerated only by tools/analyze_spot_observations.py. */
   4  
   5  import { LUX_CALIBRATION } from './config';
   6  
   7  export interface LuxCalibration {
   8    slope: number;
   9    intercept: number;
  10    enabled: boolean;
  11    /** Lower bound of the validated range; below this the intercept extrapolates unreliably. */
  12    validMinLux?: number;
  13    validMaxLux?: number;
  14  }
  15  
  16  /** Returns the lux value the engine should score with. Identity when the
  17   *  calibration is disabled. Outside the validated range (200–6000 lx) the
  18   *  large intercept over-predicts at low light, so the raw value is returned
  19   *  instead — reported as-is, not calibrated. */
  20  export function applyLuxCalibration(
  21    rawLux: number,
  22    cal: LuxCalibration = LUX_CALIBRATION,
  23  ): number {
  24    if (!cal.enabled) {
  25      return rawLux;
  26    }
  27    if (cal.validMinLux != null && rawLux < cal.validMinLux) {
  28      return Math.round(rawLux);
  29    }
  30    return Math.max(0, Math.round(cal.slope * rawLux + cal.intercept));
  31  }

   1  /** Light-suitability logic: classify the measured lux against a plant's
   2   *  (often one-sided) maintenance/preferred bands, and turn it into a [0,1]
   3   *  sub-score. Handles the sparse-threshold reality of the dataset, where many
   4   *  plants have only a maintenance floor and/or only a preferred ceiling. */
   5  
   6  import type { Plant, LightBand } from './types';
   7  
   8  /**
   9   * The lux at which a plant moves from merely surviving to thriving. Prefer an
  10   * explicit preferred floor; otherwise fall back to the next-best published
  11   * anchor so we never invent a number the dataset does not contain.
  12   */
  13  export function preferredFloor(p: Plant): number | null {
  14    return p.preferred_lux_min ?? p.maintenance_lux_max ?? p.preferred_lux_max ?? null;
  15  }
  16  
  17  export function classifyBand(lux: number, p: Plant): LightBand {
  18    const floor = p.maintenance_lux_min;
  19    if (floor != null && lux < floor) {
  20      return 'below_survival';
  21    }
  22    const ceil = p.preferred_lux_max;
  23    if (ceil != null && lux > ceil) {
  24      return 'excess';
  25    }
  26    const good = preferredFloor(p);
  27    if (good != null && lux >= good) {
  28      return 'preferred';
  29    }
  30    return 'survival';
  31  }
  32  
  33  /**
  34   * [0,1] light-fit sub-score:
  35   *   - below survival floor          -> 0    (also hard-gated elsewhere)
  36   *   - above preferred ceiling       -> 0.7  (bright excess; possible stress)
  37   *   - within/above preferred range  -> 1.0
  38   *   - between floor and "good"      -> ramps 0.6 -> 1.0
  39   */
  40  export function lightFitScore(lux: number, p: Plant): number {
  41    const floor = p.maintenance_lux_min;
  42    if (floor != null && lux < floor) {
  43      return 0;
  44    }
  45    const ceil = p.preferred_lux_max;
  46    if (ceil != null && lux > ceil) {
  47      return 0.7;
  48    }
  49    const good = preferredFloor(p);
  50    if (floor == null || good == null || good <= floor) {
  51      return 1.0;
  52    }
  53    if (lux >= good) {
  54      return 1.0;
  55    }
  56    return 0.6 + 0.4 * ((lux - floor) / (good - floor));
  57  }
  58  
  59  export function luxGapToMaintenance(lux: number, p: Plant): number | null {
  60    return p.maintenance_lux_min == null ? null : Math.round(lux - p.maintenance_lux_min);
  61  }
  62  
  63  export function luxGapToPreferred(lux: number, p: Plant): number | null {
  64    const good = preferredFloor(p);
  65    return good == null ? null : Math.round(lux - good);
  66  }

   1  /** Rule GATES — hard fails that eliminate clearly unsuitable plants before
   2   *  scoring (CLAUDE.md §2.4). Policy: lux floor + direct-sun only. Orientation
   3   *  is deliberately NOT a gate (it is a scoring factor) so the engine never
   4   *  rejects a plant whose measured spot lux is adequate. */
   5  
   6  import type { Plant, SpotInput } from './types';
   7  import { DIRECT_SUN_HOURS_THRESHOLD } from './config';
   8  
   9  /** Does the spot receive meaningful direct sun (observed, or SPA-estimated)? */
  10  export function spotHasDirectSun(spot: SpotInput): boolean {
  11    if (spot.directSunPresent === true) {
  12      return true;
  13    }
  14    if (spot.directSunHours != null && spot.directSunHours >= DIRECT_SUN_HOURS_THRESHOLD) {
  15      return true;
  16    }
  17    return false;
  18  }
  19  
  20  export interface GateResult {
  21    eliminated: boolean;
  22    reason: string | null;
  23  }
  24  
  25  export function applyGates(p: Plant, spot: SpotInput): GateResult {
  26    // GATE 1 — survival light floor: below maintenance_lux_min the plant cannot
  27    // be kept alive at this spot, by the floor's own definition.
  28    if (p.maintenance_lux_min != null && spot.lux < p.maintenance_lux_min) {
  29      return {
  30        eliminated: true,
  31        reason:
  32          `Measured light (${Math.round(spot.lux)} lux) is below ` +
  33          `${p.common_name_main}'s survival minimum (${Math.round(p.maintenance_lux_min)} lux).`,
  34      };
  35    }
  36  
  37    // GATE 2 — direct-sun incompatibility: a no-tolerance plant in a spot that
  38    // receives direct sun will scorch (per-plant sourced direct_sun_tolerance).
  39    if (p.direct_sun_tolerance === 'none' && spotHasDirectSun(spot)) {
  40      return {
  41        eliminated: true,
  42        reason: `This spot receives direct sun, which scorches ${p.common_name_main} (no direct-sun tolerance).`,
  43      };
  44    }
  45  
  46    return { eliminated: false, reason: null };
  47  }

   1  /** Weighted SCORING of the plants that survive the gates. Four thesis factors
   2   *  (light fit, direct-sun risk, distance/zone fit, evidence confidence) each
   3   *  yield a [0,1] sub-score; unavailable factors are dropped and the remaining
   4   *  weights renormalised, so partial captures (e.g. lux only, no SPA yet) still
   5   *  produce an honest score. */
   6  
   7  import type { Plant, SpotInput, FactorScore, DistanceZone, RecommendationFactors } from './types';
   8  import {
   9    WEIGHTS,
  10    DISTANCE_ZONES,
  11    LIGHT_CLASS,
  12    CONFIDENCE_SCORE,
  13    SOME_TOLERANCE_HOURS_OK,
  14  } from './config';
  15  import { lightFitScore } from './lightFit';
  16  import { spotHasDirectSun } from './gates';
  17  
  18  export function distanceZone(distanceM: number): DistanceZone {
  19    if (distanceM <= DISTANCE_ZONES.nearMaxM) {
  20      return 'near';
  21    }
  22    if (distanceM <= DISTANCE_ZONES.midMaxM) {
  23      return 'mid';
  24    }
  25    return 'deep';
  26  }
  27  
  28  export function plantLightClass(p: Plant): 'low' | 'medium' | 'high' {
  29    const m = p.maintenance_lux_min ?? 0;
  30    if (m < LIGHT_CLASS.lowMaxLux) {
  31      return 'low';
  32    }
  33    if (m < LIGHT_CLASS.mediumMaxLux) {
  34      return 'medium';
  35    }
  36    return 'high';
  37  }
  38  
  39  function lightFactor(spot: SpotInput, p: Plant): FactorScore {
  40    return {
  41      value: lightFitScore(spot.lux, p),
  42      available: true, // lux is always captured
  43      note: `Measured ${Math.round(spot.lux)} lux vs this plant's light band.`,
  44    };
  45  }
  46  
  47  function directSunFactor(spot: SpotInput, p: Plant): FactorScore {
  48    const known = spot.directSunPresent != null || spot.directSunHours != null;
  49    if (!known) {
  50      return { value: 0, available: false, note: 'No sun-path estimate captured.' };
  51    }
  52    const has = spotHasDirectSun(spot);
  53    const t = p.direct_sun_tolerance;
  54  
  55    if (t === 'tolerant') {
  56      return { value: 1.0, available: true, note: 'Tolerates direct sun.' };
  57    }
  58    if (t === 'some') {
  59      const hrs = spot.directSunHours ?? (has ? SOME_TOLERANCE_HOURS_OK + 1 : 0);
  60      const ok = !has || hrs <= SOME_TOLERANCE_HOURS_OK;
  61      return {
  62        value: ok ? 1.0 : 0.6,
  63        available: true,
  64        note: ok
  65          ? 'Tolerates the limited direct sun at this spot.'
  66          : 'Direct sun here may exceed its limited tolerance.',
  67      };
  68    }
  69    if (t === 'none') {
  70      return has
  71        ? { value: 0.2, available: true, note: 'Direct sun present; risk of leaf scorch.' }
  72        : { value: 1.0, available: true, note: 'No direct sun reaches this spot.' };
  73    }
  74    // unknown
  75    return { value: 0.6, available: true, note: 'Direct-sun tolerance unknown; treated cautiously.' };
  76  }
  77  
  78  const ZONE_CLASS_FIT: Record<DistanceZone, Record<'low' | 'medium' | 'high', number>> = {
  79    near: { high: 1.0, medium: 0.7, low: 0.4 },
  80    mid: { high: 0.7, medium: 1.0, low: 0.7 },
  81    deep: { high: 0.4, medium: 0.7, low: 1.0 },
  82  };
  83  
  84  /** Plain-language name for each distance band (used in the recommendation "why"). */
  85  const ZONE_WORDS: Record<DistanceZone, string> = {
  86    near: 'a close-up spot',
  87    mid: 'a mid-distance spot',
  88    deep: 'a spot well back from the window',
  89  };
  90  
  91  function distanceFactor(spot: SpotInput, p: Plant): FactorScore {
  92    if (spot.distanceToWindowM == null) {
  93      return { value: 0, available: false, note: 'No window distance measured.' };
  94    }
  95    const zone = distanceZone(spot.distanceToWindowM);
  96    const cls = plantLightClass(p);
  97    return {
  98      value: ZONE_CLASS_FIT[zone][cls],
  99      available: true,
 100      note: `That is ${ZONE_WORDS[zone]} for a ${cls}-light plant.`,
 101    };
 102  }
 103  
 104  function confidenceFactor(p: Plant): FactorScore {
 105    return {
 106      value: CONFIDENCE_SCORE[p.final_confidence] ?? CONFIDENCE_SCORE.low,
 107      available: true,
 108      note: `Evidence confidence: ${p.final_confidence}.`,
 109    };
 110  }
 111  
 112  export interface ScoreResult {
 113    factors: RecommendationFactors;
 114    score: number; // 0..100
 115    distanceZone: DistanceZone | null;
 116  }
 117  
 118  export function scorePlant(p: Plant, spot: SpotInput): ScoreResult {
 119    const factors: RecommendationFactors = {
 120      light: lightFactor(spot, p),
 121      directSun: directSunFactor(spot, p),
 122      distance: distanceFactor(spot, p),
 123      confidence: confidenceFactor(p),
 124    };
 125  
 126    const terms = [
 127      { f: factors.light, w: WEIGHTS.light },
 128      { f: factors.directSun, w: WEIGHTS.directSun },
 129      { f: factors.distance, w: WEIGHTS.distance },
 130      { f: factors.confidence, w: WEIGHTS.confidence },
 131    ].filter((t) => t.f.available);
 132  
 133    const wSum = terms.reduce((s, t) => s + t.w, 0);
 134    const raw = wSum > 0 ? terms.reduce((s, t) => s + t.w * t.f.value, 0) / wSum : 0;
 135  
 136    return {
 137      factors,
 138      score: Math.round(raw * 1000) / 10, // one decimal, 0..100
 139      distanceZone: spot.distanceToWindowM == null ? null : distanceZone(spot.distanceToWindowM),
 140    };
 141  }

  25  export function orientationMatch(

  40  export function displayWarning(
  41    p: Plant,
  42    factors: RecommendationFactors,
  43    orient: 'match' | 'mismatch' | 'unknown',
  44  ): string | null {
  45    const parts: string[] = [];
  46  
  47    if (factors.directSun.available && factors.directSun.value <= 0.6) {
  48      parts.push('direct-sun exposure may harm this plant');
  49    }
  50    if (p.final_confidence === 'low' || p.final_confidence === 'provisional') {
  51      parts.push(`thresholds are ${p.final_confidence}-confidence`);
  52    } else if (p.value_status && /proxy|inherit/i.test(p.value_status)) {
  53      parts.push('thresholds rely on a related-species proxy');
  54    }
  55    if (orient === 'mismatch') {
  56      parts.push('this window aspect is not the plant’s preferred orientation');
  57    }
  58  
  59    if (parts.length === 0) {
  60      return null;
  61    }
  62    return `Note: ${parts.join('; ')}.`;
  63  }

  65  /** Blend evidence confidence with measurement completeness: if optional inputs

  77  export function buildExplanation(

  79    spot: SpotInput,
  80    band: LightBand,
  81    factors: RecommendationFactors,
  82  ): string {
  83    const sentences: string[] = [];
  84  
  85    const lux = Math.round(spot.lux);
  86    const name = p.common_name_main;
  87  
  88    // Light (always present)
  89    switch (band) {
  90      case 'preferred':
  91        sentences.push(
  92          `Your spot reads ${lux} lux, which is right inside ${name}’s preferred range, so it should grow well here.`,
  93        );
  94        break;
  95      case 'survival':
  96        sentences.push(
  97          `Your spot reads ${lux} lux — enough to keep ${name} alive, but below its preferred range, so expect it to hold on rather than really thrive.`,
  98        );
  99        break;
 100      case 'excess':
 101        sentences.push(
 102          `Your spot reads ${lux} lux, brighter than ${name} likes, which can scorch or fade the leaves.`,
 103        );
 104        break;
 105      case 'below_survival':
 106        sentences.push(
 107          `Your spot reads ${lux} lux, below the minimum ${name} needs to survive, so it would slowly decline here.`,
 108        );
 109        break;
 110    }
 111  
 112    // Direct sun
 113    if (factors.directSun.available) {
 114      sentences.push(factors.directSun.note);
 115    }
 116  
 117    // Distance / zone
 118    if (factors.distance.available && spot.distanceToWindowM != null) {
 119      sentences.push(
 120        `It sits ${spot.distanceToWindowM.toFixed(1)} m from the window. ${factors.distance.note}`,
 121      );
 122    }
 123  
 124    // Orientation (informational)
 125    if (spot.windowAspect && p.aspect_orientation) {
 126      const om = orientationMatch(p, spot.windowAspect);
 127      if (om === 'match') {
 128        sentences.push(`The ${aspectWords(spot.windowAspect).toLowerCase()}-facing window is one it likes.`);
 129      } else if (om === 'mismatch') {
 130        sentences.push(
 131          `A ${aspectWords(spot.windowAspect).toLowerCase()}-facing window isn’t its usual pick, but the measured light is what really drives this match.`,
 132        );
 133      }
 134    }
 135  
 136    // Confidence
 137    sentences.push(confidenceSentence(p.final_confidence));
 138  
 139    return sentences.join(' ');
 140  }

 142  /** Plain-language version of the evidence-confidence tag. */
 143  function confidenceSentence(c: Confidence): string {
 144    switch (c) {
 145      case 'high':
 146        return 'This is backed by high-confidence, species-specific evidence.';
 147      case 'medium':
 148        return 'This is backed by medium-confidence evidence.';
 149      case 'low':
 150        return 'This rests on lower-confidence evidence, so treat it as a guide.';
 151      case 'provisional':
 152        return 'This rests on provisional evidence, so treat it as a starting point.';
 153      default:
 154        return `Evidence confidence: ${c}.`;
 155    }

   2   *  Pure function of (plants, spot); deterministic and side-effect free so it is

  11    displayLightLabel,
  12    displayWarning,
  13    orientationMatch,
  14    recommendationConfidence,
  15    buildExplanation,
  16  } from './explain';

  18  export interface RecommendResult {
  19    /** Survivors, ranked best-first. */
  20    recommended: Recommendation[];
  21    /** Gated-out plants, with the reason — kept for transparency & evaluation. */
  22    eliminated: Recommendation[];
  23  }
  24  
  25  function toRecommendation(
  26    p: Plant,
  27    spot: SpotInput,
  28    luxRaw: number,
  29  ): Recommendation {
  30    const gate = applyGates(p, spot);
  31    const band = classifyBand(spot.lux, p);
  32    const orient = orientationMatch(p, spot.windowAspect);
  33  
  34    if (gate.eliminated) {
  35      return {
  36        plant_id: p.plant_id,
  37        common_name: p.common_name_main,
  38        eliminated: true,
  39        gateReason: gate.reason,
  40        score: 0,
  41        rank: null,
  42        luxRaw,
  43        luxUsed: spot.lux,
  44        lightBand: band,
  45        distanceZone: null,
  46        factors: {
  47          light: { value: 0, available: true, note: gate.reason ?? '' },
  48          directSun: { value: 0, available: false, note: '' },
  49          distance: { value: 0, available: false, note: '' },
  50          confidence: { value: 0, available: false, note: '' },
  51        },
  52        luxGapToMaintenance: luxGapToMaintenance(spot.lux, p),
  53        luxGapToPreferred: luxGapToPreferred(spot.lux, p),
  54        orientationMatch: orient,
  55        displayLightLabel: displayLightLabel(band),
  56        displayWarning: null,
  57        recommendationConfidence: 'reduced',
  58        explanation: gate.reason ?? 'Eliminated by a rule gate.',
  59        reference: p.reference ?? null, // display-only pass-through, never scored
  60      };
  61    }
  62  
  63    const scored = scorePlant(p, spot);
  64    return {
  65      plant_id: p.plant_id,
  66      common_name: p.common_name_main,
  67      eliminated: false,
  68      gateReason: null,
  69      score: scored.score,
  70      rank: null, // assigned after sorting
  71      luxRaw,
  72      luxUsed: spot.lux,
  73      lightBand: band,
  74      distanceZone: scored.distanceZone,
  75      factors: scored.factors,
  76      luxGapToMaintenance: luxGapToMaintenance(spot.lux, p),
  77      luxGapToPreferred: luxGapToPreferred(spot.lux, p),
  78      orientationMatch: orient,
  79      displayLightLabel: displayLightLabel(band),
  80      displayWarning: displayWarning(p, scored.factors, orient),
  81      recommendationConfidence: recommendationConfidence(p, scored.factors),
  82      explanation: buildExplanation(p, spot, band, scored.factors),
  83      reference: p.reference ?? null, // display-only pass-through, never scored
  84    };
  85  }
  86  
  87  const CONFIDENCE_RANK: Record<string, number> = {
  88    high: 4,
  89    medium: 3,
  90    low: 2,
  91    provisional: 1,
  92    reduced: 0,
  93  };
  94  
  95  export function recommend(plants: Plant[], spot: SpotInput): RecommendResult {
  96    // "Both" mode: the caller passes RAW phone lux; the engine scores with the
  97    // calibrated value and every result carries both (luxRaw / luxUsed).
  98    const luxRaw = spot.lux;
  99    const effSpot: SpotInput = { ...spot, lux: applyLuxCalibration(luxRaw) };
 100    const all = plants.map((p) => toRecommendation(p, effSpot, luxRaw));
 101  
 102    const recommended = all
 103      .filter((r) => !r.eliminated)
 104      .sort(
 105        (a, b) =>
 106          b.score - a.score ||
 107          (CONFIDENCE_RANK[b.recommendationConfidence] ?? 0) -
 108            (CONFIDENCE_RANK[a.recommendationConfidence] ?? 0) ||
 109          a.common_name.localeCompare(b.common_name),
 110      );
 111    recommended.forEach((r, i) => {
 112      r.rank = i + 1;
 113    });
 114  
 115    const eliminated = all.filter((r) => r.eliminated);
 116  
 117    return { recommended, eliminated };
 118  }
